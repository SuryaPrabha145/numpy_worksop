import numpy as np

def find_smallest_divisible_by_2003():
    num = 7
    length = 1
    while num % 2003 != 0:
        num = num * 10 + 7
        length += 1
    return length

# Run the function and print the result in numpy format
length_of_smallest_divisible = find_smallest_divisible_by_2003()
np_result = np.array([length_of_smallest_divisible])
print(np_result)
